# Will make messages for specified languages in folder 'locale' (needs to be runned inside the project) where the translations can be defined
django-admin makemessages -a

# Will compile the translations
django-admin compilemessages

# Creates a .puml file from models.py file, which can be used for creation of database diagram
python3 generate_plantuml.py [path/to/models.py] > [your_file.puml]

# Creates a database diagram as .png image file (by default) [plantuml package required] 
plantuml your_file.puml
